 BlueprintExporter 第六次迭代实施计划

 Context

 前五次迭代完成了手动导出功能（Content Browser 右键、Graph 右键选中节点导出）。第六次迭代的目标是自动化：蓝图 Ctrl+S
 时自动增量导出，编辑器关闭前全量刷新，输出改为分层目录结构，并加入可配置的类型过滤器，使 ClaudeCode
 在编辑器关闭状态下可直接按需读取蓝图逻辑文件。

 #pragma once
 #include "CoreMinimal.h"
 #include "Engine/DeveloperSettings.h"
 #include "BlueprintExporterSettings.generated.h"

 UCLASS(config=EditorPerProjectUserSettings, defaultconfig,
        meta=(DisplayName="Blueprint Exporter"))
 class UBlueprintExporterSettings : public UDeveloperSettings
 {
     GENERATED_BODY()
 public:
     UBlueprintExporterSettings();
     virtual FName GetCategoryName() const override { return TEXT("Plugins"); }
     virtual FName GetSectionName() const override  { return TEXT("Blueprint Exporter"); }

     // Auto Export
     UPROPERTY(Config, EditAnywhere, Category="Auto Export")
     bool bAutoExportOnSave = false;
     UPROPERTY(Config, EditAnywhere, Category="Auto Export")
     bool bExportOnEditorClose = false;

     // Blueprint Type Filter
     UPROPERTY(Config, EditAnywhere, Category="Export Filter|Blueprint Type")
     bool bExportNormalBlueprint = true;
     UPROPERTY(Config, EditAnywhere, Category="Export Filter|Blueprint Type")
     bool bExportFunctionLibrary = true;
     UPROPERTY(Config, EditAnywhere, Category="Export Filter|Blueprint Type")
     bool bExportMacroLibrary = false;
     UPROPERTY(Config, EditAnywhere, Category="Export Filter|Blueprint Type")
     bool bExportInterface = false;
     UPROPERTY(Config, EditAnywhere, Category="Export Filter|Blueprint Type")
     bool bExportLevelScript = false;

     // Parent Class Filter
     UPROPERTY(Config, EditAnywhere, Category="Export Filter|Parent Class", meta=(AllowAbstract="true"))
     TArray<FSoftClassPath> ParentClassFilter;      // 白名单，空=不限制
     UPROPERTY(Config, EditAnywhere, Category="Export Filter|Parent Class", meta=(AllowAbstract="true"))
     TArray<FSoftClassPath> ExcludedParentClasses;  // 黑名单，优先于白名单

     // Content Filter
     UPROPERTY(Config, EditAnywhere, Category="Export Filter|Content", meta=(ClampMin=1, ClampMax=100))
     int32 MinNodeCount = 2;
 };

 新增方法实现（追加在 ExportSelectedNodesToFile 后、#undef LOCTEXT_NAMESPACE 前）：

 - SanitizeFileName — 将非字母数字字符（除 _、-）替换为 _
 - ShouldExport — 按 BlueprintType → 黑名单 → 白名单 → MinNodeCount 顺序过滤（见需求文档 §3b）
 - OnPackageSaved — 检查 bAutoExportOnSave，ForEachObjectWithPackage 找 UBlueprint，通过过滤则调用
 ExportBlueprintToCache + GenerateIndexFile
 - OnEditorPreExit — 检查 bExportOnEditorClose，调用 ExportAllBlueprints
 - ExportBlueprintToCache — Extract → 创建 {SavedDir}/BlueprintExports/{BPName}/ → 每个 Graph 写
 {SanitizedGraphName}.txt（FormatGraphOnly）→ 写 _summary.txt（FormatSummary）
 - ExportAllBlueprints — AssetRegistry 枚举所有 UBlueprint，逐个调用
 ExportBlueprintToCache，收集已导出名称集合，最后调用 GenerateIndexFile + CleanupStaleExports
 - GenerateIndexFile — 扫描 BlueprintExports/ 下所有非 _ 开头子目录，读取各 _summary.txt 提取 BP 名称/父类/Graph
 列表/变量数，写 _index.txt 和 README.md
 - CleanupStaleExports — 枚举子目录，删除不在 CurrentBPNames 集合中的目录

 FString FBlueprintTextFormatter::FormatGraphOnly(const FExportedGraph& Graph)
 {
     // SelectionContext 为空（默认），FormatGraph 以完整模式运行
     return FormatGraph(Graph);
 }

 FString FBlueprintTextFormatter::FormatSummary(const FExportedBlueprint& Blueprint)
 {
     TArray<FString> Lines;
     // 蓝图 header（与 Format() 一致）
     Lines.Add(Blueprint.ParentClass.IsEmpty()
         ? FString::Printf(TEXT("=== Blueprint: %s ==="), *Blueprint.BlueprintName)
         : FString::Printf(TEXT("=== Blueprint: %s (Parent: %s) ==="),
             *Blueprint.BlueprintName, *Blueprint.ParentClass));
     Lines.Add(TEXT(""));

     // 复用已有的 FormatVariables
     if (Blueprint.Variables.Num() > 0)
     {
         FString VarSection = FormatVariables(Blueprint.Variables);
         if (!VarSection.IsEmpty())
         {
             Lines.Add(VarSection);
             Lines.Add(TEXT(""));
         }
     }

     // Graph 列表表格（对齐，显示 node 数）
     Lines.Add(TEXT("=== Graphs ==="));
     int32 MaxNameWidth = 9;
     for (const FExportedGraph& G : Blueprint.Graphs)
         MaxNameWidth = FMath::Max(MaxNameWidth, G.GraphName.Len());

     for (const FExportedGraph& G : Blueprint.Graphs)
     {
         FString NamePad = G.GraphName + FString::ChrN(MaxNameWidth - G.GraphName.Len(), ' ');
         Lines.Add(FString::Printf(TEXT("  %s  (%s)  %d nodes"),
             *NamePad, *G.GraphType, G.Nodes.Num()));
     }
     return FString::Join(Lines, TEXT("\n")).TrimEnd();
 }

 注意事项

 - ForEachObjectWithPackage 头文件：UE5 中位于 UObject/UObjectIterator.h，若找不到可改用 GetObjectsWithPackage 或遍历
 TObjectRange<UBlueprint>
 - GetAssetsByClass API：UE5.1+ 接受 FTopLevelAssetPath（StaticClass()->GetClassPathName()）；UE5.0 接受
 FName(TEXT("Blueprint"))
 - FObjectPostSaveContext：位于 UObject/ObjectSaveContext.h，某些 UE5 版本可能在 UObject/SavePackage.h
